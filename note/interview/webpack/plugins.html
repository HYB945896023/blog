plugins 实践 #

plugin 本质是一个类或函数，通常是类，类里面必须有个 apply 方法，因为 webpack 会去调用该类实例的 apply 方法，并把compiler对象传入给我们操作，拿到compiler对象就可以调用其上面的hooks对象中的方法进行订阅

js
// 从配置中获取所有插件
// 调用每个插件上的apply方法，实现插件订阅绑定事件操作
for (const plugin of plugins) {
  // 本质就是调用compiler实例上绑定的hooks，调用tap方法进行绑定回调
  plugin.apply(compiler);
}

插件例子： #

js
module.exports = {
  // 开始编译插件
  WebpackRunPlugins: class WebpackRunPlugins {
    apply(compiler) {
      // 绑定run钩子，编译前该钩子会被调用
      compiler.hooks.run.tap("webpackRunPlugin", () => {
        console.log("开始编译");
      });
    }
  },
  // 结束编译插件
  WebpackDonePlugins: class WebpackDonePlugins {
    // 绑定done钩子，编译后该钩子会被调用
    apply(compiler) {
      compiler.hooks.run.tap("webpackDonePlugin", () => {
        console.log("结束编译");
      });
    }
  },
};
