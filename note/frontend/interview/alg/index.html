1. 二分法查找 #

思路：每次拿中间值做判断，逐步缩小范围找到目标值

  1. 先进行排序（升序)
  2. 设置两个指针start,end，以对半的方式缩小
  3. 获取中间索引的值，每次循环对比 中间索引值与目标值大小，大于目标值则说明目标值在左半区，将end修改为middle -1,反之在右半区将start改为Middle+1，直至start>=end中止寻找.
js
// 使用start,end指针
function doubleSort(arr,target){
    let start,end,element,middle
    start = 0
    end = arr.length - 1
    while(start <= end ){
        middle = Math.floor( (start + end) / 2)
        if(target > arr[middle] ){
            start = middle + 1 
        }else if(target < arr[middle]){
            end = middle - 1
        }else{
            return middle
        }
    }
    return -1
}

2. 冒泡排序 #

思路：遍历数组从左到右两两比对，小的往左移，大的右移，移动到最后一位，下次循环则忽略上次移动的最后一位，直至循环结束。

  1. 遍历数组从左到右两两比对，小的往左移，
  2. 两个循环内层循环每次循环都需要减少一次，因为最右边的元素不需要移动
js
function bubbleSort(array){
    let temp 
// 第一个for循环是确定遍历的次数，通常是长度减一
// 第二个for循环是每次循环都要减少一个，也就是最右边的不需要再进入遍历
  for (let i = 0; i < array.length - 1; i++) {
    for (let j = 0; j < array.length - 1 -i; j++) {     
        if(array[j] > array[j+1]){
            temp = array[j]
            array[j] = array[j+1]
            array[j+1] = temp       }      }   
  }  
  return array
}

3. 快速排序 #

是改进版的冒泡排序，效率高于冒泡排序

思路：选定一个基准值，遍历数组，与基准值比对，小于基准放基准值左边，大于放右边 ，下一次循环对左数组/右数组进行以上操作，递归执行。

  1. 预设一个左数组，右数组存放筛选的值
  2. 遍历对比基准值，分别放至左右数组，返回递归函数

过程模拟： [65,2,9,4,1,15]

  • 基准值：65
    left[2,9,4,1,15]（继续递归） 65 right[] (长度小于2返回) (最终1)：待定 65
  • 基准值：2
    left[1]（长度小于2返回） 2 right[9,4,15] (继续递归) (最终2)：1 2 待定
  • 基准值：9
    left[4]（长度小于2返回） 9 right[15] (长度小于2返回) (最终3)：4 9 15 往回补充结果： 1 2 4 9 15 ==> 1 2 4 9 15 65
js
function quickSort(arr){
    if( arr.length < 2 ){
        return arr
    }
    let left = []
    let right = []
    // 基准值
    let base = arr[0]
    for (let i = 1; i < arr.length; i++) {
        if(arr[i]< base){
            left.push(arr[i])
        }
        else {
            right.push(arr[i])
        }
    }  
    return [...quickSort(left),base,...quickSort(right)]
}

4. 归并排序 #

分而治之的思想

思路：遍历数组，将所有元素分成若干单一数组，将小的放在左边，下一次循环合并左右的两个数组，合并规则为判断该数组的左边与本数组左边（索引0的值）哪个小，小的放进新数组里面，再次进行比较，直至两个数组合并完成，后续重复该步骤继续合并，直至只剩下最后一个数组。

过程演推： [65,2,9,4,1,15]

  • 先拆分：[65],[2],[9],[4],[1],[15]
  • 进行归并排序：最左边长度小于2的元素直接返回[2] [2],[9,65],[4],[1,15]
  • 再次归并： [2,9,65],[1,4,15]
  • 最终归并： [1,2,4,9,15,65]
js
function mergeSort(arr){
    let len = arr.length
    if(len <= 1) return arr
    let num = Math.floor(len / 2)
    let left = mergeSort(arr.slice(0,num))
    let right = mergeSort(arr.slice(num,arr.length))
    return merge(left,right)
}
function merge(left,right){
    let [l,r] = [0,0]
    let result = []
    // 直到扫完某一方的所有元素
    while( l< left.length && r <right.length){
        if(left[l] > right[r]){
            result.push(right[r])
            r++
        }else{
            result.push(left[l])
            l++
        }
    }
        // [2，1],[5],[3,4]  第一次合并
        // [1,2][3,4,5]
        // res:[1] => (2>3?  res:[1,2] ) => ( 此时索引为3,左边为空，则直接把右边接在数组后面 )
        // 经过上面的扫荡，有一方还有剩余，则直接将该方的所有元素依次添加到result中
        result = result.concat(left.slice(l, left.length));
        result = result.concat(right.slice(r, right.length));
        console.log(result);
        return result
    }
